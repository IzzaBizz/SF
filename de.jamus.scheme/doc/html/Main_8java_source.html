00750     {
00751       boolean neg;
00752       long    ms = 0;

00754       long    percent10;
00755       String  pad;

00757       /* work with positives only */

00759       if (neg) time_val = -time_val;
00760 
00761       /* pull out seconds and ms */
00762       ms = time_val % 1000;
00763       sec = time_val / 1000;
00764 
00765       /* construct a pad to blank fill seconds out to 4 places */
00766       if (sec < 10)   
00767         pad = "   ";
00768       else if (sec < 100)  
00769         pad = "  ";
00770       else if (sec < 1000) 
00771         pad = " ";
00772       else
00773         pad = "";
00774 
00775       /* calculate 10 times the percentage of total */
00776       percent10 = (time_val*1000)/total_time;
00777 
00778       /* build and return the output string */
00779       return (neg ? "-" : "") + pad + sec + "." + 
00780              ((ms%1000)/100) + ((ms%100)/10) + (ms%10) + "sec" +
00781              " (" + percent10/10 + "." + percent10%10 + "%)";
00782     }
