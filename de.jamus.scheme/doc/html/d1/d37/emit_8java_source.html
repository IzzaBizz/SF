00727   protected static int do_escaped(PrintWriter out, char c) {
00728     StringBuffer escape = new StringBuffer();
00729     if (c <= 0xFF) {
00730       escape.append(Integer.toOctalString(c));
00731       while(escape.length() < 3) escape.insert(0, '0');
00732     } else {
00733       escape.append(Integer.toHexString(c));
00734       while(escape.length() < 4) escape.insert(0, '0');
00735       escape.insert(0, 'u');
00736     }
00737     escape.insert(0, '\\');
00738     out.print(escape.toString());

00740     // return number of bytes this takes up in UTF-8 encoding.
00741     if (c == 0) return 2;
00742     if (c >= 0x01 && c <= 0x7F) return 1;
00743     if (c >= 0x80 && c <= 0x7FF) return 2;
00744     return 3;
00745   }
